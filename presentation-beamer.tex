\section{Introduction}

\frame[t]{ % Début d'une diapositive
\frametitle{Introduction} % Titre de la diapositive
\begin{block}{Objectifs} % Titre du bloc
\begin{itemize}
\item But : déterminer si des moyennes sont égales ou non.
\item Formellement, on dispose de deux échantillons $\textbf{X}_1=(X_{1,1},\dots,X_{1,n_1})$ et $\textbf{X}_2=(X_{2,1},\dots,X_{2,n_2})$.
\item On pose $\mu_1=\E (X_{1,1})$ et $\mu_2 = \E (X_{2,1})$
\item A-t-on $\mu_1=\mu_2$ ?
\end{itemize}
\end{block}
}

\section{Aspects théoriques}

\frame[t]{\frametitle{Hypothèses}
\begin{block}{Choix des hypothèses}
$$H_0=\{\mu_1 = \mu_2 \}, \ \ H_1=\{ \mu_1 \neq \mu_2 \}$$
\end{block}
Remarques :
\begin{itemize}
\item Test bilatéral
\item De base, on suppose que les moyennes sont égales.
\item On veut éviter de dire que les moyennes sont différentes alors que ce n'est pas le cas.
\item On aurait également pu effectuer un test unilatéral via le choix de $H_0 = \{\mu_1 \leq \mu_2 \}$ et $H_1 = \{\mu_1 > \mu_2\}$ (la pertinence du choix dépend de la situation).
\end{itemize}
}

\frame[t]{\frametitle{Construction de la statistique de test}
Pour $j \in \{1,2\}$, on pose :
\begin{itemize}
\item $\mu_j = \E (X_{j,1})$
\item $\sigma_j^2 = \var (X_{j,1})$ (non connus)
\item $Z_{n_1,n_2} = \overline{X}_{1,n_1}-\overline{X}_{2,n_2}$ (différence empirique des moyennes des deux échantillons)
\end{itemize}  
\begin{block}{Normalité asymptotique}
On peut montrer que : 
$$\frac{Z_{n_1,n_2} - (\mu_1 - \mu_2)}{\sqrt{\var (Z_{n_1,n_2})}} \xrightarrow[n_1,n_2 \to \infty]{\text{loi}} \norm$$
\end{block}
\center{\textbf{Mais on ne connaît pas $\sigma_1$ et $\sigma_2$.}}
}

\begin{frame}[t]
\frametitle{Construction de la statistique de test}
C'est pourquoi on introduit les variances empiriques :
$$\forall j \in \{ 1,2 \}, \ \ \ \hat{\sigma}_{j,n_j}^2 = \frac{1}{n_j} \sum_{k=1}^{n_j} (X_{j,k} - \overline{X}_{j,n_j})^2$$
\begin{block}{Forte consistance}
On a :
$$
\frac{\sqrt{\frac{\stdone^2}{n_1} + \frac{\stdtwo^2}{n_2}}}{\sqrt{\frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2}}} \xrightarrow[n_1,n_2 \to \infty]{\text{p.s.}} 1
$$
\end{block}
\end{frame}

\begin{frame}[t]
\frametitle{Construction de la statistique de test}
\begin{block}{Région de rejet}
Avec la région de rejet
$$W_n = \left\{ Y_n:= \frac{|Z_{n_1,n_2}|}{\sqrt{\frac{\stdone^2}{n_1} + \frac{\stdtwo^2}{n_2}}} \geq \phi_{1-\alpha/2} \right\},$$
le test est consistant et de niveau asymptotique $\alpha$.
\end{block} \
\center{Rappel : $H_0=\{ \mu_1 = \mu_2 \}$ et $H_1=\{ \mu_1 \neq \mu_2 \}$}
\end{frame}

\begin{frame}[t]
\frametitle{Calcul de la p-valeur}
\begin{block}{p-valeur}
Avec $\mu=\mu_1=\mu_2$, on a :
\begin{align*}
\text{p-valeur} & = \prob_{\mu} (Y_n \geq y_n^{\text{obs}}) \\
& \simeq 2(1- F_G(y_n^{\text{obs}}))
\end{align*}
\end{block} 
où $F_G$ est la fonction de répartition de la gaussienne centrée réduite.
\end{frame}

\begin{frame}[t]
\frametitle{Similarités avec le test de Wald à un échantillon}
\begin{itemize}
\item Le test de Wald à deux échantillons généralise le test de Wald à un échantillon.
\item Pour le test de Wald à un échantillon : $H_0=\{\mu = \mu_0\}$ et $H_1= \{\mu \neq \mu_0\}$ où $\mu_0$ est fixé et connu. 
\item En prenant pour deuxième échantillon des variables aléatoires déterministes égales à $\mu_0$ dans le test de Wald à deux échantillons, on retrouve le test de Wald à un échantillon.
\end{itemize}
\end{frame}

\section{Application}

\begin{frame}[t]
\frametitle{Présentation du problème}
\begin{itemize}
\item Jeu de données : une liste d'élèves du collège dont on sait s'ils ont Internet chez eux ou pas et dont on connaît le nombre d'échecs aux examens passés.
\item Le nombre d'échecs à des examens mesure en quelque sorte les performances de l'élève.
\item Question : Les élèves sans accès à Internet réussissent-ils autant à l'école que les élèves qui ont accès à Internet ?
\end{itemize}

\end{center}
\end{frame}

\begin{frame}[t]
\frametitle{Résultats}
\begin{table}[h!]
\centering
\begin{tabular}{|c|c|c|}
\hline
\textbf{Échantillon} & \textbf{Élèves avec Internet} & \textbf{Élèves sans Internet} \\
\hline
$n$ & $n_1 = 827$ & $n_2 = 217$ \\
\hline
$\mu$ & $\mu_1 = 0.24$ & $\mu_2 = 0.36$ \\
\hline
$\sigma$ & $\sigma_1 = 0.62$ & $\sigma_2 = 0.77$ \\
\hline
\end{tabular}
\caption{Tableau des statistiques des échantillons}
\end{table}

\begin{block}{Calcul de la p-valeur}
On a :
$$\text{p-valeur} = 0.034 \text{ i.e. } 3.4 \%$$
On rejette $H_0$.
\end{block} 
\center{Rappel : $\alpha=5 \%$}
\end{frame}

\begin{frame}[t]
\frametitle{Pouvoir statistique du test}
\begin{itemize}
\item On s'intéresse au pouvoir statistique du test 
\item On cherche à calculer $\prob_{\mu_1, \mu_2} (W_n)$
\item On fixe les variances des deux échantillons et on trace le pouvoir en fonction de $\Delta \mu = \mu_1-\mu_2$
\end{itemize}
\end{frame}

\begin{frame}[t]
\frametitle{Expression du pouvoir statistique}
\begin{block}{Expression du pouvoir statistique}
Le pouvoir statistique de ce test est donné par la formule suivante :
\begin{align*}
\text{Statistical Power} &= 1 - F_G\left(\phi_{1-\alpha/2} - \frac{\Delta \mu}{\sqrt{\frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2}}}\right) \\
&\quad + F_G\left(-\phi_{1-\alpha/2} - \frac{\Delta \mu}{\sqrt{\frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2}}}\right)
\end{align*}
\noindent où $F_G$ est la fonction de répartition de la normale centrée réduite.
\end{block} 
\end{frame}

\begin{frame}[t]
\frametitle{Conclusion à tirer du test}
\begin{itemize}
\item On rejette l'hypothèse selon laquelle les élèves avec et sans Internet réussissent autant à l'école.
\item Les élèves qui n'ont pas internet réussisent moins bien.
\end{itemize}
\end{frame}

\begin{frame}[t]
\frametitle{Autre application : comparaison de proportions}
\begin{itemize}
    \item Il s'agit du cas particulier du modèle de Bernoulli.
    \item Les deux échantillons $\mathbf{X_1}$ et $\mathbf{X_2}$ suivent des lois de Bernoulli $\mathcal{B}(p_1)$ et $\mathcal{B}(p_2)$.
    \item Question du type : est-ce que $p_1 = p_2$ ? C'est-à-dire, est-ce que les deux groupes réussissent l'épreuve tout autant ?
    \item Exemples d'application :
    \begin{itemize}
        \item Effets cliniques : comparer l'efficacité de deux traitements.
        \item Marketing : évaluer le taux de clics entre deux campagnes.
        \item Éducation : comparer le taux de réussite de deux méthodes pédagogiques.
        \item Production : tester la qualité de deux lots industriels.
        \item Environnement : comparer le taux de survie de deux espèces en milieu contrôlé.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]
\frametitle{Autre application : comparaison de proportions}
\begin{itemize}
\item Exemple du cours : Neymar a réussi 70 pénaltys sur 85 (82\%) et Mbappé en a réussi 20 sur 25 (80\%)
\item Question : Neymar est-il meilleur que Mbappé au pénalty ?
\item $p_1$ (resp. $p_2$) : probabilité que Neymar marque un penalty (resp. $p_2$)
\item Avec les hypothèses $H_0=\{p_1 \leq p_2\}$ et $H_1=\{p_1>p_2\}$, $H_0$ n'est pas rejeté à 5\%
\end{itemize}
\end{frame}
